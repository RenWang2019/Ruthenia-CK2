#From Tribal to Ruthenian
province_event = {
	id = RUTH.3
	desc = EVTDESC_RUTH_3
	picture = GFX_evt_viking_throneroom_oldgods
	border = GFX_event_normal_frame_diplomacy
	
	trigger = {
		year = 867
		NOT = {
			year = 967
		}
		culture_group = east_slavic
		NOT = {
			OR = {
				culture = ruthenian
				culture = ukrainian
				culture = russian
			}
		}
	}

		months = 800
	}
	
	#ruler is tribal

		trigger = {
			NOT = {
				owner = {
					culture = ruthenian
				}
			}
		}

		trigger = {
			owner = {

		}
